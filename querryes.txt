
//Cria a tabela de usuarios com ID_user de chave primaria
CREATE TABLE Usuarios
(
    ID_user integer PRIMARY KEY AUTOINCREMENT,
    usuario varchar(25) NOT NULL,
    senha varchar(15),
);


// Cria a tabela da hora do trabalhador e usa o id_user da tabela usuario como chave estrangeira
CREATE TABLE horaTrabalhador
(
    ID_Carro integer PRIMARY KEY AUTO_INCREMENT,
    dataAtual date,
    horaEntrada time,
    horaSaida time,
    Justificativa varchar(255),
    ID_user integer,
    CONSTRAINT fk_UsuHora FOREIGN KEY (ID_user) REFERENCES Pessoa (ID_user)
);
